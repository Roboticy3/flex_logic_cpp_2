debug (more compiler output, gdb-debuggable with msys):
  scons use_mingw=yes debug_symbols=yes dev_build=yes separate_debug_symbols=yes

quick (faster compile):
  mkdir ./cache/quick
  scons dev_build=no compiledb=True cache_path=.cache/quick cache_limit=2

  gdb --args .\bin\godot.windows.editor.dev.x86_64.exe ..\..\moon-science\project.godot

  gdb --args .\bin\godot.windows.editor.dev.x86_64.exe --path ..\..\moon-science mockup\mockup_grid.tscn

  .\bin\godot.windows.editor.dev.x86_64.exe ..\..\moon-science\project.godot
